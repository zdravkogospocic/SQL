create table jobs (
job_id integer PRIMARY KEY,
job_title nvarchar(25),
min_salary integer,
max_salary integer
)

CREATE table employees (
employee_id integer PRIMARY KEY,
 first_name nvarchar(25),
 last_name nvarchar(25),
 email nvarchar(25),
 phone_number char(9),
 hire_date date,
 job_employee_id integer,
 salary integer,
  manager_id integer,
  department_id integer,
 constraint jobs_employee_2_jobs
 FOREIGN KEY (job_employee_id) REFERENCES jobs(job_id) 
  )
  
  CREATE table dependens (
    dependent_id integer PRIMARY KEY,
    first_name nvarchar(25),
    last_name nvarchar(25),
    relationship nvarchar(25),
    d_e_id integer
)

CREATE table departments (
department_id integer primary key,
 department_name nvarchar(25),
 location_id integer,
  CONSTRAINT departments_2_employee
  FOREIGN KEY(location_id) REFERENCES employees(employee_id)
)

create table regions (
  region_id integer PRIMARY KEY,
  region_name nvarchar(25)
  )
  
  CREATE table countries (
  country_id integer,
  country_name nvarchar(25),
   country_region_id integer,
   CONSTRAINT c_2_region 
   FOREIGN KEY (country_region_id) REFERENCES regions(region_id)
    )
    
    
  CREATE table locations (
  location_id integer PRIMARY KEY,
   street_address nvarchar(25),
    postal_code nvarchar(25),
    city nvarchar(25),
    state_province nvarchar(25),
    location_country_id integer
    )
